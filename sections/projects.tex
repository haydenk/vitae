\section{Projects}

\entry{\href{https://github.com/haydenk/overseer}{overseer}}{Open Source}{}{2026}
\begin{itemize}
  \item Zero-dependency Procfile-based process manager with colored output, formation support,
        and graceful shutdown — a Go clone of Foreman.
\end{itemize}
\vspace{0.75em}

\entry{\href{https://github.com/haydenk/homestead}{homestead}}{Open Source}{}{2026}
\begin{itemize}
  \item Self-hosted personal dashboard for home lab environments with real-time health checks,
        service organization, search, and keyboard shortcuts. Built with Go stdlib and vanilla JS.
\end{itemize}
\vspace{0.75em}

\entry{\href{https://github.com/haydenk/gridsync}{gridsync}}{Open Source}{}{2026}
\begin{itemize}
  \item Automated data pipeline ingesting 3--5M ERCOT energy records into PostgreSQL with a
        fuzzy search API built on Axum, SQLx, and pg\_trgm.
\end{itemize}
\vspace{0.75em}

\entry{\href{https://github.com/haydenk/tablekit}{tablekit}}{Open Source}{}{2026}
\begin{itemize}
  \item Framework-agnostic data table library with zero runtime dependencies using a
        plugin architecture for composable, minimal table functionality.
\end{itemize}
\vspace{0.75em}

\entry{\href{https://github.com/haydenk/pizza-party}{pizza-party}}{Open Source}{}{2026}
\begin{itemize}
  \item Provider-agnostic C client library and CLI for pizza delivery APIs, supporting store
        search, menu retrieval, and order tracking via a virtual table plugin architecture.
  \item A fun attempt to recreate an application from college.
\end{itemize}
