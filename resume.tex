% ---------- commands ----------
\newcommand{\resumeheader}[5]{%
  \begin{center}
    {\huge\bfseries\color{accent} #1}\\[1em]
    \small
    \faEnvelope\ \href{mailto:#2}{#2}
    \quad\textbar\quad
    \faLinkedin\ \href{https://linkedin.com/in/#4}{linkedin.com/in/#4}\\[0.2em]
    \faPhone\ #3
    \quad\textbar\quad
    \faGithub\ \href{https://github.com/#5}{github.com/#5}
  \end{center}
}

\newcommand{\entry}[4]{%
  % #1 = title/role  #2 = org  #3 = location  #4 = date range
  \noindent
  \textbf{#1} \hfill \textit{\small #4}\\
  \textit{\small #2} \hfill \textit{\small #3}
  \vspace{0.1em}
}

\newcommand{\entryskills}[2]{%
  % #1 = label  #2 = value
  \noindent\textbf{#1:} #2\\[0.15em]
}
